> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Rebuild the QueryVeda SQL practice platform from vanilla HTML/CSS/JS into a Next.js 14 static-export app with TypeScript, Tailwind CSS, shadcn/ui, CodeMirror 6, PGlite, and Supabase auth — mobile-first and deployable to GitHub Pages.

**Architecture:** Next.js App Router with `output: 'export'` for static generation. All SQL execution happens client-side via PGlite WASM. Auth and progress sync via Supabase. Theme via next-themes + Tailwind dark mode. Component library via shadcn/ui.

**Tech Stack:** Next.js 14, TypeScript, Tailwind CSS v4, shadcn/ui, CodeMirror 6, PGlite (WASM), Supabase JS SDK, next-themes

```typescript
export interface AuthUser {

  rawUser: User | null;
  loading: boolean;
  login: (email: string, password: string) => Promise<{ ok: boolean; msg?: string }>;
  signup: (name: string, email: string, password: string) => Promise<{ ok: boolean; msg?: string }>;

  logout: () => Promise<void>;

export const AuthContext = createContext<AuthContextValue | null>(null);
```

- [ ] **Step 3: Create AuthProvider**

Create `queryveda/components/auth/auth-provider.tsx`:

```tsx
"use client";

import { useEffect, useState, useCallback } from "react";
import type { User } from "@supabase/supabase-js";
import { supabase } from "@/lib/supabase";
import { AuthContext, type AuthUser } from "@/hooks/use-auth";

function toAuthUser(user: User | null): AuthUser | null {
  if (!user) return null;
  const meta = user.user_metadata || {};
  return {
    id: user.id,
    name: meta.full_name || meta.name || user.email?.split("@")[0] || "User",
    email: user.email,
    avatar: meta.avatar_url || meta.picture || null,
  };
}

export function AuthProvider({ children }: { children: React.ReactNode }) {
  const [rawUser, setRawUser] = useState<User | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    supabase.auth.getSession().then(({ data: { session } }) => {
      setRawUser(session?.user ?? null);
      setLoading(false);
    });

    const { data: { subscription } } = supabase.auth.onAuthStateChange(
      (_event, session) => {
        setRawUser(session?.user ?? null);
      }
    );

    return () => subscription.unsubscribe();
  }, []);

  const login = useCallback(async (email: string, password: string) => {
    if (!email || !password) return { ok: false, msg: "Email and password are required." };
    const { error } = await supabase.auth.signInWithPassword({ email, password });
    if (error) return { ok: false, msg: error.message };
    return { ok: true };
  }, []);

  const signup = useCallback(async (name: string, email: string, password: string) => {
    if (!name || !email || !password) return { ok: false, msg: "All fields are required." };
    if (password.length < 6) return { ok: false, msg: "Password must be at least 6 characters." };
    const { data, error } = await supabase.auth.signUp({
      email,
      password,
      options: { data: { full_name: name } },
    });
    if (error) return { ok: false, msg: error.message };
    if (data.user && !data.session) {
      return { ok: true, msg: "Check your email to confirm your account." };
    }
    return { ok: true };
  }, []);

  const loginWithGoogle = useCallback(async () => {
    await supabase.auth.signInWithOAuth({
      provider: "google",
      options: { redirectTo: window.location.origin + (process.env.NEXT_PUBLIC_BASE_PATH || "") + "/" },
    });
  }, []);

  const loginWithLinkedIn = useCallback(async () => {
    await supabase.auth.signInWithOAuth({
      provider: "linkedin_oidc",
      options: { redirectTo: window.location.origin + (process.env.NEXT_PUBLIC_BASE_PATH || "") + "/" },
    });
  }, []);

  const logout = useCallback(async () => {
    await supabase.auth.signOut();
  }, []);

  return (
    <AuthContext.Provider
      value={{
        user: toAuthUser(rawUser),
        rawUser,
        loading,
        login,
        signup,
        loginWithGoogle,
        loginWithLinkedIn,
        logout,
      }}
    >
      {children}
    </AuthContext.Provider>
  );
}
```

```typescript
import type { Question } from "./types";

// Format a Date for display/comparison
function fmtDate(v: Date): string {
  const p = (n: number) => String(n).padStart(2, "0");
  const Y = v.getFullYear(), M = p(v.getMonth() + 1), D = p(v.getDate());
  const h = v.getHours(), mi = v.getMinutes(), s = v.getSeconds();
  if (h || mi || s) return `${Y}-${M}-${D} ${p(h)}:${p(mi)}:${p(s)}`;
  return `${Y}-${M}-${D}`;
}

export function displayValue(v: unknown): string | null {
  if (v === null || v === undefined) return null;
  if (v instanceof Date) return fmtDate(v);
  return String(v);
}

function normalize(v: unknown): string {
  if (v === null || v === undefined) return "\u2205";
  let s = v instanceof Date ? fmtDate(v) : String(v);
  s = s.trim();
  if (/^-?\d+(\.\d+)?$/.test(s)) return String(Math.round(parseFloat(s) * 10000) / 10000);
  return s.toLowerCase();
}

export interface CompareResult {
  pass: boolean;
  msg: string;
}

export function compareResults(
  expected: { cols: string[]; rows: unknown[][] },
  actual: { cols: string[]; rows: unknown[][] }
): CompareResult {
  const ec = expected.cols.length;
  if (actual.rows.length && actual.rows[0].length !== ec)
    return { pass: false, msg: `Column count mismatch: expected ${ec} column(s), got ${actual.rows[0].length}. Expected order: ${expected.cols.join(", ")}.` };
  if (actual.rows.length === 0 && expected.rows.length > 0)
    return { pass: false, msg: "Your query returned 0 rows." };
  if (actual.rows.length !== expected.rows.length)
    return { pass: false, msg: `Row count mismatch: expected ${expected.rows.length}, got ${actual.rows.length}.` };
  const key = (r: unknown[]) => r.map(normalize).join("\u0001");
  const e = expected.rows.map(key).sort();
  const a = actual.rows.map(key).sort();
  for (let i = 0; i < e.length; i++) {
    if (e[i] !== a[i]) return { pass: false, msg: "Values don't match the expected output (compared independent of row order)." };
  }
  return { pass: true, msg: "All rows match. \u2714" };
}

export interface QueryResult {
  cols: string[];
  rows: unknown[][];
}

export async function executeQuery(db: any, sql: string): Promise<QueryResult> {
  const res = await db.query(sql);
  const cols = res.fields.map((f: any) => f.name);
  const rows = res.rows.map((r: any) => cols.map((c: string) => r[c]));
  return { cols, rows };
}

export async function runTests(
  db: any,
  question: Question,
  sql: string
): Promise<{ passed: boolean; message: string; userResult: QueryResult | null }> {
  // Setup and run visible test
  try {
    await db.exec(question.setup);
  } catch (e: any) {
    return { passed: false, message: `Setup error: ${e.message}`, userResult: null };
  }

  let userResult: QueryResult;
  try {
    userResult = await executeQuery(db, sql);
  } catch (e: any) {
    return { passed: false, message: `SQL error: ${e.message}`, userResult: null };
  }

  const visibleResult = compareResults(
    { cols: question.cols, rows: question.rows },
    userResult
  );

  if (!visibleResult.pass) {
    return { passed: false, message: visibleResult.msg, userResult };
  }

  // Run hidden tests
  const hiddenTests = question.tests || [];
  for (let i = 0; i < hiddenTests.length; i++) {
    const tc = hiddenTests[i];
    try {
      await db.exec(tc.setup);
    } catch (e: any) {
      return { passed: false, message: `Setup error in hidden test: ${e.message}`, userResult };
    }
    let tcResult: QueryResult;
    try {
      tcResult = await executeQuery(db, sql);
    } catch (e: any) {
      return { passed: false, message: `SQL error on hidden test case ${i + 2}: ${e.message}`, userResult };
    }
    const tcCompare = compareResults({ cols: question.cols, rows: tc.rows }, tcResult);
    if (!tcCompare.pass) {
      return {
        passed: false,
        message: `Passed visible test but failed hidden test case ${i + 2}. ${tcCompare.msg} Your query may be hardcoded for the sample data.`,
        userResult,
      };
    }
  }

  // Reset tables to visible test data
  try {
    await db.exec(question.setup);
  } catch {}

  const totalTests = 1 + hiddenTests.length;
  return {
    passed: true,
    message: `Correct! All ${totalTests} test case${totalTests > 1 ? "s" : ""} passed.`,
    userResult,
  };
}
```

```typescript
"use client";

import { useEffect } from "react";

export function useProtection() {
  useEffect(() => {
    const handleContextMenu = (e: MouseEvent) => e.preventDefault();

    const handleKeyDown = (e: KeyboardEvent) => {
      // Block Ctrl+Shift+I, Ctrl+Shift+J, Ctrl+U, F12
      if (
        (e.ctrlKey && e.shiftKey && (e.key === "I" || e.key === "J")) ||
        (e.ctrlKey && e.key === "u") ||
        e.key === "F12"
      ) {
        e.preventDefault();
      }
    };

    const handleCopy = (e: ClipboardEvent) => {
      const target = e.target as HTMLElement;
      // Allow copy inside CodeMirror editor
      if (target.closest(".cm-editor")) return;
      e.preventDefault();
    };

    document.addEventListener("contextmenu", handleContextMenu);
    document.addEventListener("keydown", handleKeyDown);
    document.addEventListener("copy", handleCopy);

    // Disable text selection except in editor
    const style = document.createElement("style");
    style.textContent = `
      body { -webkit-user-select: none; user-select: none; }
      .cm-editor, .cm-editor * { -webkit-user-select: text; user-select: text; }
    `;
    document.head.appendChild(style);

    return () => {
      document.removeEventListener("contextmenu", handleContextMenu);
      document.removeEventListener("keydown", handleKeyDown);
      document.removeEventListener("copy", handleCopy);
      style.remove();
    };
  }, []);
}
```

```typescript
"use client";

import { useState, useMemo } from "react";
import type { Difficulty, Topic, Question } from "@/lib/types";
import { DIFFICULTY_ORDER } from "@/lib/constants";

export function useFilters(questions: Question[]) {
  const [difficulty, setDifficulty] = useState<Difficulty | "All">("All");
  const [topic, setTopic] = useState<Topic | "All">("All");
  const [search, setSearch] = useState("");

  const filtered = useMemo(() => {
    let result = questions;
    if (difficulty !== "All") result = result.filter((q) => q.difficulty === difficulty);
    if (topic !== "All") result = result.filter((q) => q.topic === topic);
    if (search.trim()) {
      const s = search.toLowerCase();
      result = result.filter((q) => q.title.toLowerCase().includes(s));
    }
    return result.sort((a, b) => {
      const di = DIFFICULTY_ORDER[a.difficulty] - DIFFICULTY_ORDER[b.difficulty];
      if (di !== 0) return di;
      return a.id - b.id;
    });
  }, [questions, difficulty, topic, search]);

  return { difficulty, setDifficulty, topic, setTopic, search, setSearch, filtered };
}
```

```tsx
import { useEffect, useRef } from "react";
import { EditorView, keymap, placeholder as placeholderExt } from "@codemirror/view";
import { EditorState } from "@codemirror/state";
import { sql, PostgreSQL } from "@codemirror/lang-sql";
import { defaultKeymap, indentWithTab } from "@codemirror/commands";
import { oneDark } from "@codemirror/theme-one-dark";
import { autocompletion } from "@codemirror/autocomplete";
import { useTheme } from "next-themes";

interface SQLEditorProps {
  initialValue: string;
  onChange: (value: string) => void;
  onRun: () => void;
  tables?: Record<string, string[]>;
}

export function SQLEditor({ initialValue, onChange, onRun, tables }: SQLEditorProps) {
  const containerRef = useRef<HTMLDivElement>(null);
  const viewRef = useRef<EditorView | null>(null);
  const { theme } = useTheme();

  useEffect(() => {
    if (!containerRef.current) return;

    const runKeymap = keymap.of([
      {
        key: "Ctrl-Enter",
        run: () => { onRun(); return true; },
      },
      {
        key: "Cmd-Enter",
        run: () => { onRun(); return true; },
      },
    ]);

    const state = EditorState.create({
      doc: initialValue,
      extensions: [
        runKeymap,
        keymap.of([...defaultKeymap, indentWithTab]),
        sql({ dialect: PostgreSQL, tables: tables || {} }),
        autocompletion(),
        placeholderExt("-- write your PostgreSQL query here\nSELECT ..."),
        EditorView.updateListener.of((update) => {
          if (update.docChanged) {
            onChange(update.state.doc.toString());
          }
        }),
        EditorView.theme({
          "&": { height: "220px", fontSize: "14px" },
          ".cm-scroller": { fontFamily: "var(--font-mono), ui-monospace, monospace" },
        }),
        ...(theme === "dark" ? [oneDark] : []),
      ],
    });

    const view = new EditorView({ state, parent: containerRef.current });
    viewRef.current = view;

    return () => view.destroy();
  }, [theme]); // Recreate on theme change

  // Update doc when initialValue changes (question navigation)
  useEffect(() => {
    const view = viewRef.current;
    if (view && view.state.doc.toString() !== initialValue) {
      view.dispatch({
        changes: { from: 0, to: view.state.doc.length, insert: initialValue },
      });
    }
  }, [initialValue]);

  return (
    <div ref={containerRef} className="rounded-lg border overflow-hidden" />
  );
}
```

```tsx
import { useRef, useCallback } from "react";

interface SplitPaneProps {
  left: React.ReactNode;
  right: React.ReactNode;
}

export function SplitPane({ left, right }: SplitPaneProps) {
  const containerRef = useRef<HTMLDivElement>(null);

  const handleMouseDown = useCallback((e: React.MouseEvent) => {
    e.preventDefault();
    const container = containerRef.current;
    if (!container) return;

    const startX = e.clientX;
    const leftEl = container.querySelector("[data-pane='left']") as HTMLElement;
    const startWidth = leftEl.getBoundingClientRect().width;

    const onMove = (ev: MouseEvent) => {
      const newW = Math.max(200, Math.min(startWidth + ev.clientX - startX, window.innerWidth - 300));
      container.style.gridTemplateColumns = `${newW}px 6px 1fr`;
    };

    const onUp = () => {
      document.removeEventListener("mousemove", onMove);
      document.removeEventListener("mouseup", onUp);
    };

    document.addEventListener("mousemove", onMove);
    document.addEventListener("mouseup", onUp);
  }, []);

  return (
    <>
      {/* Desktop: split pane */}
      <div
        ref={containerRef}
        className="hidden md:grid h-[calc(100vh-8rem)]"
        style={{ gridTemplateColumns: "400px 6px 1fr" }}
      >
        <div data-pane="left" className="overflow-auto p-5 bg-card">
          {left}
        </div>
        <div
          className="cursor-col-resize bg-border hover:bg-primary transition-colors"
          onMouseDown={handleMouseDown}
        />
        <div className="overflow-auto p-5 flex flex-col gap-3">
          {right}
        </div>
      </div>

      {/* Mobile: stacked */}
      <div className="flex flex-col md:hidden">
        <div className="border-b p-4">{left}</div>
```

```tsx
"use client";

import { use, useEffect, useState, useCallback, useRef } from "react";
import { useRouter } from "next/navigation";
import { Button } from "@/components/ui/button";
import { questions, getQuestionById, getSortedQuestions } from "@/lib/questions";
import { runTests, executeQuery } from "@/lib/pglite";
import { usePGlite } from "@/hooks/use-pglite";
import { useStorage } from "@/hooks/use-storage";
import { SQLEditor } from "@/components/practice/sql-editor";
import { ProblemPanel } from "@/components/practice/problem-panel";
import { ResultTable } from "@/components/practice/result-table";
import { HintsPanel } from "@/components/practice/hints-panel";
import { SolutionPanel } from "@/components/practice/solution-panel";
import { SplitPane } from "@/components/practice/split-pane";
import type { Question } from "@/lib/types";

export function generateStaticParams() {
  return questions.map((q) => ({ id: String(q.id) }));
}

export default function PracticePage({ params }: { params: Promise<{ id: string }> }) {
  const { id } = use(params);
  const router = useRouter();
  const { db, ready, error: pgliteError } = usePGlite();
  const { getStatus, markSolved, markAttempted, getSavedSQL, saveSQL } = useStorage();

  const question = getQuestionById(Number(id));
  const sortedQs = getSortedQuestions();
  const currentIndex = sortedQs.findIndex((q) => q.id === Number(id));

  const [sql, setSQL] = useState("");
  const [verdict, setVerdict] = useState<{ type: "pass" | "fail" | "error" | ""; message: string }>({ type: "", message: "" });
  const [userResult, setUserResult] = useState<{ cols: string[]; rows: unknown[][] } | null>(null);
  const [schemaTables, setSchemaTables] = useState<Array<{ name: string; cols: string[]; rows: unknown[][] }>>([]);
  const [showSolution, setShowSolution] = useState(false);
  const [running, setRunning] = useState(false);
  const sqlRef = useRef(sql);
  sqlRef.current = sql;

  // Load saved SQL and setup tables when question or db changes
  useEffect(() => {
    if (!question || !db || !ready) return;
    const saved = getSavedSQL(question.id);
    setSQL(saved);
    setVerdict({ type: "", message: "" });
    setUserResult(null);
    setShowSolution(false);

    (async () => {
      try {
        await db.exec(question.setup);
        const tables: typeof schemaTables = [];
        for (const t of question.tables) {
          const result = await executeQuery(db, `SELECT * FROM ${t}`);
          tables.push({ name: t, ...result });
        }
        setSchemaTables(tables);
      } catch (e: any) {
        setSchemaTables([]);
      }
    })();
  }, [question?.id, db, ready]);

  const handleChange = useCallback((value: string) => {
    setSQL(value);
    if (question) {
      saveSQL(question.id, value);
      if (value.trim()) markAttempted(question.id);
    }
  }, [question]);

  const handleRun = useCallback(async () => {
    if (!question || !db || running) return;
    const currentSQL = sqlRef.current.trim().replace(/;\s*$/, "");
    if (!currentSQL) {
      setVerdict({ type: "error", message: "Write a query first." });
      return;
    }

    setRunning(true);
    setVerdict({ type: "", message: "Running..." });
    setUserResult(null);

    const result = await runTests(db, question, currentSQL);
    setRunning(false);

    if (result.passed) {
      markSolved(question.id);
      setVerdict({ type: "pass", message: result.message });
    } else {
      if (result.userResult) markAttempted(question.id);
      setVerdict({ type: result.userResult ? "fail" : "error", message: result.message });
    }
    setUserResult(result.userResult);
    setShowSolution(true);
  }, [question, db, running]);

  if (!question) {
    return (
      <div className="flex min-h-[60vh] items-center justify-center">
        <p className="text-muted-foreground">Question not found.</p>
      </div>
    );
  }

  const tables = schemaTables.reduce<Record<string, string[]>>((acc, t) => {
    acc[t.name] = t.cols;
    return acc;
  }, {});

  const leftPanel = (
    <ProblemPanel question={question} schemaTables={schemaTables} />
  );

  const rightPanel = (
    <>
      {pgliteError ? (
        <div className="rounded-lg bg-[var(--qv-danger-bg)] p-4 text-sm text-[var(--qv-danger)]">
          Failed to start PostgreSQL: {pgliteError}
        </div>
      ) : !ready ? (
        <p className="text-sm text-muted-foreground">Starting PostgreSQL...</p>
      ) : null}

      <div className="flex items-center gap-2 flex-wrap">
        <Button onClick={handleRun} disabled={!ready || running}>
          {running ? "Running..." : "Run & Check \u25B6"}
        </Button>
        <Button
          variant="outline"
          onClick={() => {
            setSQL("");
            if (question) saveSQL(question.id, "");
            setVerdict({ type: "", message: "" });
            setUserResult(null);
          }}
        >
          Clear
        </Button>
        <span className="ml-auto text-xs text-muted-foreground">Cmd+Enter to run</span>
      </div>

      <HintsPanel hints={question.hints} />

      <SQLEditor
        initialValue={sql}
        onChange={handleChange}
        onRun={handleRun}
        tables={tables}
      />

      {verdict.type && (
        <div
          className={`rounded-lg p-3 text-sm font-semibold ${
            verdict.type === "pass"
              ? "bg-[var(--qv-success-bg)] text-[var(--qv-success)]"
              : verdict.type === "fail"
              ? "bg-[var(--qv-danger-bg)] text-[var(--qv-danger)]"
              : "bg-[var(--qv-warning-bg)] text-[var(--qv-warning)]"
          }`}
        >
          {verdict.message}
        </div>
      )}

      {userResult && (
        <div>
          <p className="mb-1 text-xs font-semibold uppercase tracking-wider text-muted-foreground">
            Your Output
          </p>
          <ResultTable cols={userResult.cols} rows={userResult.rows} />
        </div>
      )}

      {verdict.type === "fail" && (
        <div>
          <p className="mb-1 text-xs font-semibold uppercase tracking-wider text-muted-foreground">
            Expected Output
          </p>
          <ResultTable cols={question.cols} rows={question.rows} />
        </div>
      )}

      {showSolution && <SolutionPanel question={question} passed={verdict.type === "pass"} />}
    </>
  );

  return (
    <div>
      {/* Top bar */}
      <div className="flex items-center gap-3 border-b px-4 py-2 flex-wrap">
        <span className="text-sm font-bold">{question.title}</span>
        <div className="ml-auto flex items-center gap-2">
          <Button
            variant="outline"
            size="sm"
            disabled={currentIndex <= 0}
            onClick={() => router.push(`/practice/${sortedQs[currentIndex - 1]?.id}`)}
          >
            Prev
          </Button>
          <Button
            variant="outline"
            size="sm"
            disabled={currentIndex >= sortedQs.length - 1}
            onClick={() => router.push(`/practice/${sortedQs[currentIndex + 1]?.id}`)}
          >
            Next
          </Button>
        </div>
      </div>

      <SplitPane left={leftPanel} right={rightPanel} />
    </div>
  );
}
```

```tsx
import { TOPICS, TOPIC_COLORS } from "@/lib/constants";
import type { Topic } from "@/lib/types";

interface SkillRadarProps {
  byTopic: Record<Topic, { solved: number; total: number }>;
}

export function SkillRadar({ byTopic }: SkillRadarProps) {
  const cx = 150, cy = 150, r = 110;
  const n = TOPICS.length;

  const angles = TOPICS.map((_, i) => (Math.PI * 2 * i) / n - Math.PI / 2);

  const gridPoints = (scale: number) =>
    angles.map((a) => `${cx + r * scale * Math.cos(a)},${cy + r * scale * Math.sin(a)}`).join(" ");

  const dataPoints = TOPICS.map((topic, i) => {
    const pct = byTopic[topic].total === 0 ? 0 : byTopic[topic].solved / byTopic[topic].total;
    return `${cx + r * pct * Math.cos(angles[i])},${cy + r * pct * Math.sin(angles[i])}`;
  }).join(" ");

  return (
    <div className="flex justify-center">
      <svg viewBox="0 0 300 300" className="w-full max-w-[300px]">
        {/* Grid lines */}
        {[0.25, 0.5, 0.75, 1].map((scale) => (
          <polygon key={scale} points={gridPoints(scale)} fill="none" stroke="currentColor" strokeOpacity={0.1} />
        ))}
        {/* Axis lines */}
        {angles.map((a, i) => (
          <line key={i} x1={cx} y1={cy} x2={cx + r * Math.cos(a)} y2={cy + r * Math.sin(a)} stroke="currentColor" strokeOpacity={0.1} />
        ))}
        {/* Data polygon */}
        <polygon points={dataPoints} fill="rgb(var(--primary))" fillOpacity={0.2} stroke="rgb(var(--primary))" strokeWidth={2} />
        {/* Labels */}
        {TOPICS.map((topic, i) => {
          const labelR = r + 20;
          const x = cx + labelR * Math.cos(angles[i]);
          const y = cy + labelR * Math.sin(angles[i]);
```

```tsx
import { useMemo } from "react";

      <SkillRadar byTopic={stats.byTopic} />
```

## Task 11: Leaderboard Page

**Files:**
- Create: `queryveda/app/leaderboard/page.tsx`

- [ ] **Step 1: Create Leaderboard page**

Create `queryveda/app/leaderboard/page.tsx`:

```tsx
"use client";

import { useEffect, useState } from "react";
import { Card } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { supabase } from "@/lib/supabase";
import { useAuth } from "@/hooks/use-auth";

interface LeaderboardEntry {
  user_id: string;
  display_name: string;
  avatar_url: string | null;
  solved_count: number;
  streak: number;
  completion_pct: number;
}

type TimePeriod = "all" | "month" | "week";

export default function LeaderboardPage() {
  const { user } = useAuth();
  const [entries, setEntries] = useState<LeaderboardEntry[]>([]);
  const [period, setPeriod] = useState<TimePeriod>("all");
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    async function fetchLeaderboard() {
      setLoading(true);
      try {
        // Query user_progress grouped by user
        let query = supabase
          .from("user_progress")
          .select("user_id, status, solved_at");

        if (period === "month") {
          const since = new Date();
          since.setMonth(since.getMonth() - 1);
          query = query.gte("solved_at", since.toISOString());
        } else if (period === "week") {
          const since = new Date();
          since.setDate(since.getDate() - 7);
          query = query.gte("solved_at", since.toISOString());
        }

        const { data } = await query.eq("status", "solved");

        if (!data) { setEntries([]); setLoading(false); return; }

        // Aggregate by user
        const userMap = new Map<string, { solved: number; dates: Set<string> }>();
        for (const row of data) {
          if (!userMap.has(row.user_id)) {
            userMap.set(row.user_id, { solved: 0, dates: new Set() });
          }
          const u = userMap.get(row.user_id)!;
          u.solved++;
          if (row.solved_at) u.dates.add(row.solved_at.slice(0, 10));
        }

        const leaderboard: LeaderboardEntry[] = Array.from(userMap.entries()).map(
          ([uid, data]) => ({
            user_id: uid,
            display_name: uid.slice(0, 8) + "...",
            avatar_url: null,
            solved_count: data.solved,
            streak: data.dates.size,
            completion_pct: Math.round((data.solved / 75) * 100),
          })
        );

        leaderboard.sort((a, b) => b.solved_count - a.solved_count);
        setEntries(leaderboard);
      } catch (e) {
        console.warn("Leaderboard fetch failed:", e);
      }
      setLoading(false);
    }

    fetchLeaderboard();
  }, [period]);

  return (
    <div className="mx-auto max-w-3xl px-4 py-8">
      <h1 className="mb-6 text-3xl font-bold">Leaderboard</h1>

      <div className="mb-6 flex gap-2">
        {(["all", "month", "week"] as TimePeriod[]).map((p) => (
          <Button
            key={p}
            variant={period === p ? "default" : "outline"}
            size="sm"
            onClick={() => setPeriod(p)}
          >
            {p === "all" ? "All Time" : p === "month" ? "This Month" : "This Week"}
          </Button>
        ))}
      </div>

      {loading ? (
        <p className="text-muted-foreground">Loading...</p>
      ) : entries.length === 0 ? (
        <Card className="p-8 text-center text-muted-foreground">
          No data yet. Be the first to solve problems!
        </Card>
      ) : (
        <div className="overflow-x-auto rounded-lg border">
          <table className="w-full text-sm">
            <thead>
              <tr className="border-b bg-muted/50">
                <th className="px-4 py-3 text-left font-medium w-16">Rank</th>
                <th className="px-4 py-3 text-left font-medium">User</th>
                <th className="px-4 py-3 text-center font-medium">Solved</th>
                <th className="px-4 py-3 text-center font-medium">Active Days</th>
                <th className="px-4 py-3 text-center font-medium">Completion</th>
              </tr>
            </thead>
            <tbody>
              {entries.map((entry, i) => (
                <tr
                  key={entry.user_id}
                  className={`border-b transition-colors ${
                    user?.id === entry.user_id ? "bg-primary/10" : "hover:bg-muted/30"
                  }`}
                >
                  <td className="px-4 py-3 font-bold">{i + 1}</td>
                  <td className="px-4 py-3 flex items-center gap-2">
                    {entry.avatar_url && (
                      <img src={entry.avatar_url} alt="" className="w-6 h-6 rounded-full" />
                    )}
                    <span>{entry.display_name}</span>
                    {user?.id === entry.user_id && (
                      <span className="text-xs text-primary font-medium">(you)</span>
                    )}
                  </td>
                  <td className="px-4 py-3 text-center">{entry.solved_count}</td>
                  <td className="px-4 py-3 text-center">{entry.streak}</td>
                  <td className="px-4 py-3 text-center">{entry.completion_pct}%</td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      )}
    </div>
  );
}
```

```yaml
name: Deploy to GitHub Pages

on:
  push:
    branches: [main]

  contents: read
  pages: write
  id-token: write

concurrency:

          cache: npm

      - run: npm ci
      - run: npm run build
        env:
          NEXT_PUBLIC_SUPABASE_URL: ${{ secrets.NEXT_PUBLIC_SUPABASE_URL }}
          NEXT_PUBLIC_SUPABASE_ANON_KEY: ${{ secrets.NEXT_PUBLIC_SUPABASE_ANON_KEY }}
          NEXT_PUBLIC_BASE_PATH: /sql_compiler
      - run: touch out/.nojekyll
      - run: cp out/404/index.html out/404.html || true
      - uses: actions/upload-pages-artifact@v3
        with:
          path: queryveda/out

  deploy:
    environment:
      name: github-pages
      url: ${{ steps.deployment.outputs.page_url }}
    runs-on: ubuntu-latest
    needs: build
    steps:
      - id: deployment
        uses: actions/deploy-pages@v4
```
